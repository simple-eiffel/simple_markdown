note
	description: "[
				Simple Markdown - Production-ready Markdown to HTML conversion for Eiffel.
		
				CommonMark compliant with GitHub Flavored Markdown (GFM) and extended syntax.
		
				== CommonMark Core ==
				- Headings (# through ######)
				- Bold (**text** or __text__)
				- Italic (*text* or _text_)
				- Inline code (code)
				- Code blocks (``` with language hints)
				- Links [text](url)
				- Images ![alt](url)
				- Unordered lists (-, *, +)
				- Ordered lists (1. 2. 3.)
				- Blockquotes (>)
				- Horizontal rules (---, ***, ___)
				- Paragraphs
		
				== GFM Extensions ==
				- Strikethrough (~~text~~)
				- Tables (| col | col |)
				- Task lists (- [ ] and - [x])
				- Autolinks (URLs automatically linked)
		
				== Extended Syntax ==
				- Footnotes ([^1] references)
				- Highlight (==text==)
				- Subscript (~text~) / Superscript (^text^)
				- Heading IDs (# Heading {#custom-id})
				- Table of contents generation
		
				Usage:
					create md.make
					html := md.to_html ("# Hello World")
		
					-- Parse file
					html := md.to_html_from_file ("README.md")
		
					-- Get table of contents
					toc := md.table_of_contents
	]"

class interface
	SIMPLE_MARKDOWN

create 
	make

feature -- Conversion

	to_html (a_markdown: STRING_8): STRING_8
			-- Convert a_markdown to HTML.
		require
			markdown_not_void: a_markdown /= Void
		ensure
			last_html_set: last_html.same_string (Result)

	to_html_from_file (a_path: STRING_8): STRING_8
			-- Convert markdown file at a_path to HTML.
		require
			path_not_void: a_path /= Void
			path_not_empty: not a_path.is_empty

	to_html_fragment (a_markdown: STRING_8): STRING_8
			-- Convert a_markdown inline elements only (no block wrapping).
		require
			markdown_not_void: a_markdown /= Void
	
feature -- Access

	last_html: STRING_8
			-- Result of last conversion.

	headings: ARRAYED_LIST [TUPLE [level: INTEGER_32; text: STRING_8; id: detachable STRING_8]]
			-- Headings found during last parse (for TOC generation).

	table_of_contents: STRING_8
			-- Generate HTML table of contents from last parse.
	
feature -- Query

	is_heading (a_line: STRING_8): BOOLEAN
			-- Is a_line a heading?
		require
			line_not_void: a_line /= Void

	is_blockquote (a_line: STRING_8): BOOLEAN
			-- Is a_line a blockquote?
		require
			line_not_void: a_line /= Void

	is_unordered_list_item (a_line: STRING_8): BOOLEAN
			-- Is a_line an unordered list item?
		require
			line_not_void: a_line /= Void

	is_task_list_item (a_line: STRING_8): BOOLEAN
			-- Is a_line a task list item (- [ ] or - [x])?
		require
			line_not_void: a_line /= Void

	is_ordered_list_item (a_line: STRING_8): BOOLEAN
			-- Is a_line an ordered list item?
		require
			line_not_void: a_line /= Void

	is_code_fence (a_line: STRING_8): BOOLEAN
			-- Is a_line a code fence?
		require
			line_not_void: a_line /= Void

	is_horizontal_rule (a_line: STRING_8): BOOLEAN
			-- Is a_line a horizontal rule?
		require
			line_not_void: a_line /= Void

	is_blank_line (a_line: STRING_8): BOOLEAN
			-- Is a_line blank?
		require
			line_not_void: a_line /= Void
