note
	description: "Processes inline Markdown elements (bold, italic, links, etc.)"
	author: "Larry Rix"
	date: "$Date$"
	revision: "$Revision$"

class interface
	MD_INLINE_PROCESSOR

create 
	make

feature -- Processing

	process (a_text: STRING_8): STRING_8
			-- Process all inline markdown elements in a_text.
		require
			text_not_void: a_text /= Void
	
feature -- Element Processing

	process_bold (a_text: STRING_8): STRING_8
			-- Convert **text** and __text__ to <strong>.
		require
			text_not_void: a_text /= Void

	process_italic (a_text: STRING_8): STRING_8
			-- Convert *text* and _text_ to <em>.
		require
			text_not_void: a_text /= Void

	process_strikethrough (a_text: STRING_8): STRING_8
			-- Convert ~~text~~ to <del> (GFM).
		require
			text_not_void: a_text /= Void

	process_code_inline (a_text: STRING_8): STRING_8
			-- Convert code to <code>.
		require
			text_not_void: a_text /= Void

	process_links (a_text: STRING_8): STRING_8
			-- Convert [text](url) to <a href="url">text</a>.
		require
			text_not_void: a_text /= Void

	process_images (a_text: STRING_8): STRING_8
			-- Convert ![alt](url) to <img>.
		require
			text_not_void: a_text /= Void

	process_autolinks (a_text: STRING_8): STRING_8
			-- Convert bare URLs to links (GFM autolinks).
		require
			text_not_void: a_text /= Void
	
feature -- Extended Syntax Processing

	process_highlight (a_text: STRING_8): STRING_8
			-- Convert ==text== to <mark> (highlighted text).
		require
			text_not_void: a_text /= Void

	process_superscript (a_text: STRING_8): STRING_8
			-- Convert ^text^ to <sup> (superscript).
		require
			text_not_void: a_text /= Void

	process_subscript (a_text: STRING_8): STRING_8
			-- Convert ~text~ to <sub> (subscript).
			-- Note: Must be careful not to conflict with ~~strikethrough~~.
		require
			text_not_void: a_text /= Void

	process_footnote_refs (a_text: STRING_8): STRING_8
			-- Convert [^id] to footnote reference link.
		require
			text_not_void: a_text /= Void
